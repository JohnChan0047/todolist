### 3.1、添加task
1、存储逻辑：对form进行监控》如果有输入，就添加到task_list中》更新本地存储store
```
/*监控任务添加按钮*/
    function listen_add_task() {
        $form_add_task.on('submit', function (e) {
            var new_task = {}, $input;
            e.preventDefault();
            $input = $(this).find('input[name=content]');
            new_task.content = $input.val();
            if(!new_task.content) return;
            if(add_task(new_task)){
                $input.val('');
            }
        })
    }

    /*添加任务*/
    function add_task(new_task) {
        task_list.push(new_task);
        refresh_task_list();
        return true;
    }

    /*更新存储列表*/
    function refresh_task_list() {
        store.set('task_list', task_list);
    }
```
2、页面更新逻辑：把task_list中每一项task_item制作成html》然后添加到hrml的`<div class="tasks-list"> </div>`中
```
/*展示task_list*/
    function render_task_list() {
        var $task_list = $('.tasks-list');
        $task_list.html('');
        for(var i = 0; i < task_list.length; i++){
            var item = task_list[i];
            if(item)

                $task_list.prepend($task)

    function render_task_item(data, index) {
        if(index === undefined || !data) return;
        var list_item_tpl =
            '<div class="task-item" data-index="' + index + '">' +
            '<span><input class="complete" '+ (data.complete ? 'checked' : '') + ' type="checkbox"></span>' +
            '<span class="task-content">' + data.content + '</span>' +
            '<span class="fr">' +
            '<span class="action delete"> 删除</span>' +
            '<span class="action detail"> 详情</span>' +
            '</span>' +
            '</div>';
        return $(list_item_tpl);
    }
```
### 3.2、删除task
1、删除逻辑：对删除按键进行监控》如果有点击，作是否删除确认》删除
先声明全局变量`$delete_task`  
然后在`render_task_list()`中进行定义和监控`$delete_task = $('.action.delete');
        listen_delete_task();`
```
function listen_delete_task() {
        $delete_task.on('click', function () {
            var $this = $(this);
            var index = $this.parent().parent().data('index');
            var tmp = confirm('确定删除？');
            tmp ? delete_task(index) : null;
        })
    }
    /*删除任务*/
function delete_task(index) {
        if(index === undefined || !task_list[index]) return;
        delete task_list[index];
        refresh_task_list();
    }
```

### 3.4、完成打勾的分类
逻辑：监控打勾》改变complete的布尔值》在展示列表时分类展示
```
/*监控打勾*/
    function listen_checkbox_complete() {
        $checkbox_complete.on('click', function () {
            var $this = $(this);
            var index = $this.parent().parent().data('index');
            var item = get(index);
            if(item.complete)
                update_task(index, {complete: false});
            else
                update_task(index, {complete: true});
        })
    }
    /*展示task_list*/
    function render_task_list() {
        var $task_list = $('.tasks-list');
        $task_list.html('');
        var complete_list = [];
        for(var i = 0; i < task_list.length; i++){
            var item = task_list[i];
            if(item && item.complete)
                complete_list[i] = item;
            else {
                var $task = render_task_item(item, i);
                $task_list.prepend($task)
            }
        }

        for(var j = 0; j < complete_list.length; j++){

            var $task = render_task_item(complete_list[j], j);
            if(!$task) continue;
            $task.addClass('complete');
            $task_list.append($task);
        }

        $delete_task = $('.action.delete');
        $detail_task = $('.action.detail');
        $task_item = $('.task-item');
        $checkbox_complete = $('.complete');
        listen_delete_task();
        listen_detail_task();
        listen_checkbox_complete();
    }
```

```
/*检查时间是否到*/
    function task_remind_check() {
        var current_timestamp;
        var itl = setInterval(function () {
            for(var i = 0; i < task_list.length; i++){
                var item = get(i), task_timestamp;
                if(!item || !item.remind_date || item.informed || item.complete) continue;
                current_timestamp = (new Date()).getTime();
                task_timestamp = (new Date(item.remind_date)).getTime();
                if(task_timestamp - current_timestamp <= 1 ){
                    update_task(i, {informed: true})
                    show_msg(item.content);
                }
            }
        }, 300);
    }

    /*展示提示*/
    function show_msg(msg) {
        if(!msg) return;
        $msg_content.html(msg);
        $alert.get(0).play();
        $msg.show();
    }
```
